class Outer {

    void display() {
        System.out.println("inside outerMethod");
    }

    class Inner {
        void display() {
            System.out.println("In a nested class method");
        }
    }
}

public class Point {
    public static void main(String[] args) {

        Outer.Inner in = new Outer().new Inner();
        in.display();

        Outer out = new Outer();
        out.display();
    }
}
